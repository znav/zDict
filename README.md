-INSTALLATION



1. git clone git@github.com:znav/zDict.git
2. cp zDict/dictionary ~/bin/
3. cp zDict/dictionary_autocomplete /etc/bash_completion.d/
4. . ~/bin/dictionary
5. . /etc/bash_completion.d/dictionary_autocomplete
